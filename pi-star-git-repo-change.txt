sudo git remote -v

sudo git remote set-url origin https://github.com/erstec/Pi-Star_v4_Binaries_Bin.git

sudo git remote set-url origin https://github.com/AndyTaylorTweet/Pi-Star_v4_Binaries_Bin.git
